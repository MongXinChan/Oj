[27. 移除元素 - 力扣（LeetCode）](https://leetcode.cn/problems/remove-element/?envType=study-plan-v2&envId=top-interview-150)

利用模拟指针的方法来写。

```cpp
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int src=0,dst=0;
        while(src<nums.size()){
            if(nums[src]==val){
                src++;
            }else{
                nums[dst++]=nums[src++];
            }
        }
        return dst;
    }
};
```

删除有序数组中的重复项

```cpp
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int sz=nums.size();
        if(sz==0){
            return 0;
        }
        int fast=1,slow=1;
        while(fast<sz){
            if(nums[fast]!=nums[fast-1]){
                nums[slow++]=nums[fast];
            }
            fast++;
        }
        return slow;
    }
};
```
